<# .SYNOPSIS Clears the Intune Management Extension cache and restarts the IME service. Forces Intune to re-evaluate all Win32 app detection and compliance state. .NOTES Must be run as an elevated (local admin) account. Use when: app shows as installed in Intune but install/update is not executing on device. #> # --- Elevation check --- if (-not ([Security.Principal.WindowsPrincipal] [Security.Principal.WindowsIdentity]::GetCurrent()).IsInRole([Security.Principal.WindowsBuiltInRole]::Administrator)) { Write-Error "This script must be run as Administrator. Right-click PowerShell and select 'Run as administrator'." exit 1 } $imeCachePaths = @( "C:\ProgramData\Microsoft\IntuneManagementExtension\GRS", "C:\ProgramData\Microsoft\IntuneManagementExtension\GRS2", "C:\ProgramData\Microsoft\IntuneManagementExtension\StateStore", "C:\ProgramData\Microsoft\IntuneManagementExtension\Cache" ) # --- Stop IME service --- Write-Output "Stopping IntuneManagementExtension service..." try { Stop-Service -Name IntuneManagementExtension -Force Write-Output "Service stopped." } catch { Write-Error "Failed to stop service: $_" exit 1 } # --- Clear cache folders --- Write-Output "" Write-Output "Clearing IME cache folders..." foreach ($path in $imeCachePaths) { if (Test-Path $path) { try { Remove-Item -Path $path -Recurse -Force Write-Output " Cleared: $path" } catch { Write-Warning " Could not clear: $path -- $_" } } else { Write-Output " Not present (skipped): $path" } } # --- Start IME service --- Write-Output "" Write-Output "Starting IntuneManagementExtension service..." try { Start-Service -Name IntuneManagementExtension Write-Output "Service started." } catch { Write-Error "Failed to start service: $_" exit 1 } Write-Output "" Write-Output "Done. Intune will re-evaluate app compliance on next check-in." Write-Output "To trigger immediately: open Company Portal and click Sync."
